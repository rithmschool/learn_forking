#create a key pair, use git terminal
ssh-keygen -t rsa -b 4096 -C "PUT YOUR EMAIL HERE"
Enter a file in which to save the key (/Users/you/.ssh/id_rsa): [Press enter]
You will then be prompted to enter a passphrase [Press enter]
again [Press enter]
#start ssh-agent manually
eval "$(ssh-agent -s)"
#add key
ssh-add ~/.ssh/id_rs
#paste key in clipboard
